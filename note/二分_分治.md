#### lc69 sqrt(x)

public int mySqrt(int x) {
        if (x == 0) {
            return 0;
        }
        int low = 1;
        int high = x;

        while (low <= high) {
            int mid = low + (high - low) / 2;
            int sqrt = x / mid;
            if (sqrt == mid) {
                return mid;
            }else if (sqrt > mid) {
                low = mid + 1;
            }else {
                high = mid - 1;
            }
        }
        return high;
    }
